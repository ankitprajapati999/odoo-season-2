-- Copy and paste these statements into the Supabase SQL Editor.

-- 1. VEHICLES TABLE
CREATE TABLE IF NOT EXISTS public.vehicles (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    type TEXT NOT NULL CHECK (type IN ('Truck', 'Van', 'Mini', 'Bus', 'SUV', 'Sedan')),
    capacity TEXT NOT NULL,
    region TEXT NOT NULL,
    fuel_efficiency TEXT NOT NULL,
    status TEXT NOT NULL CHECK (status IN ('Active', 'In Shop', 'Off Duty', 'Maintenance')),
    license_plate TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable RLS on public.vehicles
ALTER TABLE public.vehicles ENABLE ROW LEVEL SECURITY;

-- Select policies
CREATE POLICY "Allow public read access to vehicles" 
ON public.vehicles FOR SELECT USING (true);

-- Write policies (for authorized users)
CREATE POLICY "Allow authenticated users to write vehicles" 
ON public.vehicles FOR ALL TO authenticated USING (true);


-- 2. DRIVERS TABLE
CREATE TABLE IF NOT EXISTS public.drivers (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    license_number TEXT NOT NULL,
    safety_score INTEGER DEFAULT 100 CHECK (safety_score BETWEEN 0 AND 100),
    status TEXT NOT NULL CHECK (status IN ('Active', 'Off Duty', 'On Leave')),
    phone TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable RLS on public.drivers
ALTER TABLE public.drivers ENABLE ROW LEVEL SECURITY;

-- Select policies
CREATE POLICY "Allow public read access to drivers" 
ON public.drivers FOR SELECT USING (true);

-- Write policies
CREATE POLICY "Allow authenticated users to write drivers" 
ON public.drivers FOR ALL TO authenticated USING (true);


-- 3. TRIPS TABLE
CREATE TABLE IF NOT EXISTS public.trips (
    id TEXT PRIMARY KEY,
    vehicle_id TEXT REFERENCES public.vehicles(id) ON DELETE SET NULL,
    driver_id TEXT REFERENCES public.drivers(id) ON DELETE SET NULL,
    status TEXT NOT NULL CHECK (status IN ('On Time', 'Delayed', 'Completed', 'Active')),
    start_time TIMESTAMP WITH TIME ZONE NOT NULL,
    end_time TIMESTAMP WITH TIME ZONE,
    region TEXT NOT NULL,
    route_name TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable RLS on public.trips
ALTER TABLE public.trips ENABLE ROW LEVEL SECURITY;

-- Select policies
CREATE POLICY "Allow public read access to trips" 
ON public.trips FOR SELECT USING (true);

-- Write policies
CREATE POLICY "Allow authenticated users to write trips" 
ON public.trips FOR ALL TO authenticated USING (true);


-- 4. MAINTENANCE LOGS TABLE
CREATE TABLE IF NOT EXISTS public.maintenance (
    id TEXT PRIMARY KEY,
    vehicle_id TEXT REFERENCES public.vehicles(id) ON DELETE CASCADE,
    issue TEXT NOT NULL,
    priority TEXT NOT NULL CHECK (priority IN ('Low', 'Medium', 'High', 'Critical')),
    status TEXT NOT NULL CHECK (status IN ('Pending', 'In Progress', 'Completed')),
    cost NUMERIC DEFAULT 0.0,
    date DATE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable RLS on public.maintenance
ALTER TABLE public.maintenance ENABLE ROW LEVEL SECURITY;

-- Select policies
CREATE POLICY "Allow public read access to maintenance" 
ON public.maintenance FOR SELECT USING (true);

-- Write policies
CREATE POLICY "Allow authenticated users to write maintenance" 
ON public.maintenance FOR ALL TO authenticated USING (true);


-- 5. EXPENSES TABLE
CREATE TABLE IF NOT EXISTS public.expenses (
    id TEXT PRIMARY KEY,
    vehicle_id TEXT REFERENCES public.vehicles(id) ON DELETE CASCADE,
    type TEXT NOT NULL CHECK (type IN ('Fuel', 'Tolls', 'Maintenance', 'Insurance', 'Other')),
    amount NUMERIC DEFAULT 0.0,
    description TEXT,
    date DATE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable RLS on public.expenses
ALTER TABLE public.expenses ENABLE ROW LEVEL SECURITY;

-- Select policies
CREATE POLICY "Allow public read access to expenses" 
ON public.expenses FOR SELECT USING (true);

-- Write policies
CREATE POLICY "Allow authenticated users to write expenses" 
ON public.expenses FOR ALL TO authenticated USING (true);
